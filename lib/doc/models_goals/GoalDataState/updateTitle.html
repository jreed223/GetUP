Future<void> updateTitle(String goalId) async {
  for (dynamic goal in goals) {
    if (goal.goalId == goalId) {
      await FirebaseFirestore.instance
          .collection('Users')
          .doc(FirebaseAuth.instance.currentUser!.uid)
          .collection('goals')
          .doc(goalId)
          .update({'title': goal.title});
    }
  }
}
